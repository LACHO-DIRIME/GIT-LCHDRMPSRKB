// ══ DEFINICIÓN BACKUP DOOR ════════════════════════════════════
// {door} en WORK = vault · acceso restringido · CRYPTO (key seat) certifica
// BACKUP/door = todo backup DRA pasa por el vault protegido
// No es backup automático descuidado · es MuSave soberano con {door}
// Protocolo: SAMU @ audita → {door} abre → backup sellado → {door} cierra

// ══ TIPOS DE BACKUP ═══════════════════════════════════════════

                              //rsync → hub GitHub + drive soberano

                              //MU_STATE+CONTEXT+IDENTITY preservados

UNICODE: BKP_THEATER          //THEATER/ + RUNNERS/ · macros soberanos
UNICODE: BKP_CONFIG           //IMV/config/ + .sovereign · identidad sistema
UNICODE: BKP_FULL             //backup completo · tar DRA/ → sellado H63

// ══ PROTOCOLO DOOR ════════════════════════════════════════════
UNICODE: DOOR_OPEN            //WORK {door} abre · SAMU @ autoriza · CRYPTO key
                              //WORK {door} =><= .. abre .. vault_backup_autorizado
UNICODE: DOOR_WRITE           //escribe backup en vault · STACKING UF[H52]
UNICODE: DOOR_SEAL            //CRYPTO (spark seat) firma el backup · [NdO]
UNICODE: DOOR_CLOSE           //WORK {door} cierra · vault sellado · H63
UNICODE: DOOR_VERIFY          //verifica integridad · hash backup · FABRIC_TX
UNICODE: DOOR_RESTORE         //restauración soberana · SAMU @ autoriza · MuRestore
