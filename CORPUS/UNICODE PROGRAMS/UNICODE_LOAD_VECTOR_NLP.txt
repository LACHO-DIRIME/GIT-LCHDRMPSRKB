// ══ DEFINICIÓN ════════════════════════════════════════════════
// LOAD VECTOR NLP = protocolo de carga semántica
// Convierte texto natural en vectores LACHO routing
// Pipeline: texto → tokenizar → clasificar → route → sentencia
// Módulo: core/language_routing.py · IME{[(!!ROUTE!!)]}
// Resultado: RoutedSentence con library/subject/verb/object/knot

// ══ PIPELINE NLP→LACHO ════════════════════════════════════════
UNICODE: NLP_INPUT       //texto natural crudo · KU · pre-procesamiento · .blue
UNICODE: NLP_TOKENIZE    //tokenización · split · normalización · lowercase
UNICODE: NLP_CLASSIFY    //clasificación biblioteca · TRUST/WORK/SAMU/...
UNICODE: NLP_ROUTE       //language_routing.route() · IME{[(!!ROUTE!!)]}
UNICODE: NLP_OUTPUT      //RoutedSentence · LACHO verificable · pre-WU
UNICODE: NLP_VALIDATE    //grammar.validate() · VALID/WARNING/INVALID
UNICODE: NLP_SCORE       //lacho_score() · Scalar S · 0.0→1.0
UNICODE: NLP_STACK       //ledger.record() · cristaliza WU · TX++

// ══ VECTORES DE CARGA ═════════════════════════════════════════
// Cada biblioteca tiene su vector semántico de activación:

UNICODE: VEC_TRUST       //vector TRUST · keywords: verifica/declara/autoriza
                         //trigger: contrato · fundación · validar · certificar
                         //→ SUJETOS: FOUNDATION/[scope]/[term]/[command]
UNICODE: VEC_WORK        //vector WORK · keywords: ejecuta/procesa/activa
                         //trigger: hacer · correr · procesar · construir
                         //→ SUJETOS: {actuator}/{brake}/{doorman-mobile}
UNICODE: VEC_SAMU        //vector SAMU · keywords: audita/revisa/evalúa
                         //trigger: revisar · controlar · supervisar · moderar
                         //→ SUJETOS: @ / #rise / #set / $mon→$sun
UNICODE: VEC_ACTIVITY    //vector ACTIVITY · keywords: inicia/lanza/cristaliza
                         //trigger: comenzar · empezar · abrir · activar ciclo
                         //→ SUJETOS: UF[H01]→UF[H64] · hexagrama
UNICODE: VEC_CRYPTO      //vector CRYPTO · keywords: certifica/protege/firma
                         //trigger: seguridad · wallet · clave · blockchain
                         //→ SUJETOS: (key seat)/(shield seat)/(spark seat)
UNICODE: VEC_SOCIAL      //vector SOCIAL · keywords: conecta/transmite/distribuye
                         //trigger: red · comunidad · WeChat · relay · difundir
                         //→ SUJETOS: {relay}/{launch-bot}/{headcat}
UNICODE: VEC_GATE        //vector GATE · keywords: filtra/bloquea/espera
                         //trigger: acceso · permiso · control · latencia
                         //→ SUJETOS: UF[H05]/UF[H06]/UF[H56]
UNICODE: VEC_STACKING    //vector STACKING · keywords: cristaliza/ancla/preserva
                         //trigger: guardar · archivar · inmutabilizar · consolidar
                         //→ SUJETOS: UF[H52]/UF[H48]/UF[H26]
UNICODE: VEC_METHOD      //vector METHOD · keywords: calcula/modela/deriva
                         //trigger: calcular · analizar · estadística · fórmula
                         //→ SUJETOS: <equation>/<operator_flow>/<stat_onto>

// ══ OPERADORES DE CARGA ═══════════════════════════════════════
UNICODE: LOAD(INPUT)     //carga texto natural → buffer KU · .blue
UNICODE: LOAD(VECTOR)    //proyecta texto → espacio vectorial LACHO
UNICODE: LOAD(LIBRARY)   //clasifica biblioteca dominante · score por lib
UNICODE: LOAD(SUBJECT)   //selecciona sujeto canónico · SUJETOS_CANONICOS
UNICODE: LOAD(VERB)      //selecciona verbo natural · VERBOS_NATURALES
UNICODE: LOAD(OBJECT)    //extrae objeto semántico · entidad principal
UNICODE: LOAD(KNOT)      //asigna nudo · KNOT_DEFAULT por biblioteca
UNICODE: LOAD(VALIDATE)  //grammar.validate() · score threshold 0.75
UNICODE: LOAD(EMIT)      //emite sentencia LACHO · pre-STACK · WU ready

// ══ IME ROUTING ═══════════════════════════════════════════════
UNICODE: IME_ROUTE       //IME{[(!!ROUTE!!)]} · NLP→LACHO via language_routing
UNICODE: IME_SCORE       //score de clasificación · confianza del route
UNICODE: IME_FALLBACK    //fallback TRUST si clasificación ambigua
UNICODE: IME_BATCH       //route_batch() · múltiples textos simultáneos
UNICODE: IME_HISTORY     //historial de routes · suggest_from_history()
