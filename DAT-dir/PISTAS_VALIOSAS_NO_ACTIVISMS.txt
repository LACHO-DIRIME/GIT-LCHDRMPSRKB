function detectar_acronimo_individual(palabra, contexto_sistema):
    # Paso 1: Identificar individualidad fundamental
    individualidad_fundamental = identificar_individualidad_fundamental(palabra)
    
    if not individualidad_fundamental:
        return {"veredicto": "NO_ES_ACRÓNIMO", "confianza": 0.1, "razon": "sin_individualidad_fundamental"}
    
    # Paso 2: Evaluar coincidencias
    coincidencias = evaluar_coincidencias_individual(individualidad_fundamental, contexto_sistema)
    
    # Paso 3: Validar estructura
    estructura_valida = validar_estructura_individual(palabra, coincidencias)
    
    # Paso 4: Buscar aplicación de individualidad
    aplicacion = buscar_aplicacion_individual(palabra)
    
    # Paso 5: Calcular confianza
    confianza = calcular_confianza_individual(individualidad_fundamental, coincidencias, estructura_valida, aplicacion)
    
    # Paso 6: Generar veredicto
    veredicto = generar_veredicto_individual(confianza)
    
    return {
        "palabra": palabra,
        "individualidad_fundamental": individualidad_fundamental,
        "coincidencias": coincidencias,
        "estructura_valida": estructura_valida,
        "aplicacion": aplicacion,
        "confianza": confianza,
        "veredicto": veredicto
    }
