function detectar_acronimo_completo(palabra, contexto_sistema):
    # Método 1: Detectar por letras raras
    resultado_letras = detectar_acronimo_por_letras_raras(palabra, contexto_sistema)
    
    # Método 2: Detectar por elementos no alfabéticos
    resultado_no_alfabeticos = detectar_acronimo_no_alfabetico(palabra, contexto_sistema)
    
    # Método 3: Detectar por estructuras complejas
    resultado_estructuras = detectar_acronimo_estructural(palabra, contexto_sistema)
    
    # Método 4: Detectar por unidades conceptuales
    resultado_conceptuales = detectar_acronimo_conceptual(palabra, contexto_sistema)
    
    # Método 5: Detectar por unidades fundamentales
    resultado_fundamentales = detectar_acronimo_existencial(palabra, contexto_sistema)
    
    # Método 6: Detectar por contextos universales
    resultado_universales = detectar_acronimo_trascendente(palabra, contexto_sistema)
    
    # Método 7: Detectar por estados existenciales
    resultado_existenciales = detectar_acronimo_existencial(palabra, contexto_sistema)
    
    # Método 8: Detectar por procesos permanentes
    resultado_permanentes = detectar_acronimo_permanente(palabra, contexto_sistema)
    
    # Método 9: Detectar por estados autónomos
    resultado_autonomos = detectar_acronimo_autonomo(palabra, contexto_sistema)
    
    # Método 10: Detectar por instrucciones inherentes
    resultado_instructivos = detectar_acronimo_instructivo(palabra, contexto_sistema)
    
    # Método 11: Detectar por comunicaciones universales
    resultado_comunicativos = detectar_acronimo_comunicativo(palabra, contexto_sistema)
    
    # Método 12: Detectar por ambientes fundamentales
    resultado_ambientales = detectar_acronimo_ambiental(palabra, contexto_sistema)
    
    # Método 13: Detectar por componentes fundamentales
    resultado_componentes = detectar_acronimo_componente(palabra, contexto_sistema)
    
    # Método 14: Detectar por conocimientos fundamentales
    resultado_conocimientos = detectar_acronimo_conocimiento(palabra, contexto_sistema)
    
    # Método 15: Detectar por fenómenos fundamentales
    resultado_fenomenologicos = detectar_acronimo_fenomenologico(palabra, contexto_sistema)
    
    # Método 16: Detectar por procesos operativos
    resultado_ejecutivos = detectar_acronimo_ejecutivo(palabra, contexto_sistema)
    
    # Método 17: Detectar por operaciones estratégicas
    resultado_estrategicos = detectar_acronimo_estrategico(palabra, contexto_sistema)
    
    # Método 18: Detectar por liderazgo fundamental
    resultado_liderazgo = detectar_acronimo_liderazgo(palabra, contexto_sistema)
    
    # Método 19: Detectar por realización fundamental
    resultado_realizacion = detectar_acronimo_realizacion(palabra, contexto_sistema)
    
    # Método 20: Detectar por organización estructural
    resultado_estructural = detectar_acronimo_estructural(palabra, contexto_sistema)
    
    # Método 21: Detectar por confiabilidad fundamental
    resultado_confiable = detectar_acronimo_confiable(palabra, contexto_sistema)
    
    # Método 22: Detectar por individualidad fundamental
    resultado_individual = detectar_acronimo_individual(palabra, contexto_sistema)
    
    # Combinar evidencias
    confianza_combinada = max(
        resultado_letras.confianza, 
        resultado_no_alfabeticos.confianza, 
        resultado_estructuras.confianza,
        resultado_conceptuales.confianza,
        resultado_fundamentales.confianza,
        resultado_universales.confianza,
        resultado_existenciales.confianza,
        resultado_permanentes.confianza,
        resultado_autonomos.confianza,
        resultado_instructivos.confianza,
        resultado_comunicativos.confianza,
        resultado_ambientales.confianza,
        resultado_componentes.confianza,
        resultado_conocimientos.confianza,
        resultado_fenomenologicos.confianza,
        resultado_ejecutivos.confianza,
        resultado_estrategicos.confianza,
        resultado_liderazgo.confianza,
        resultado_realizacion.confianza,
        resultado_estructural.confianza,
        resultado_confiable.confianza,
        resultado_individual.confianza
    )
    
    # Evidencia combinada
    if confianza_combinada > 0.8:
        veredicto = "ACRÓNIMO_CONFIRMADO"
    elif confianza_combinada > 0.5:
        veredicto = "INVESTIGACIÓN_REQUERIDA"
    else:
        veredicto = "NO_ES_ACRÓNIMO"
    
    return {
        "palabra": palabra,
        "resultado_letras": resultado_letras,
        "resultado_no_alfabeticos": resultado_no_alfabeticos,
        "resultado_estructuras": resultado_estructuras,
        "resultado_conceptuales": resultado_conceptuales,
        "resultado_fundamentales": resultado_fundamentales,
        "resultado_universales": resultado_universales,
        "resultado_existenciales": resultado_existenciales,
        "resultado_permanentes": resultado_permanentes,
        "resultado_autonomos": resultado_autonomos,
        "resultado_instructivos": resultado_instructivos,
        "resultado_comunicativos": resultado_comunicativos,
        "resultado_ambientales": resultado_ambientales,
        "resultado_componentes": resultado_componentes,
        "resultado_conocimientos": resultado_conocimientos,
        "resultado_fenomenologicos": resultado_fenomenologicos,
        "resultado_ejecutivos": resultado_ejecutivos,
        "resultado_estrategicos": resultado_estrategicos,
        "resultado_liderazgo": resultado_liderazgo,
        "resultado_realizacion": resultado_realizacion,
        "resultado_estructural": resultado_estructural,
        "resultado_confiable": resultado_confiable,
        "resultado_individual": resultado_individual,
        "confianza_combinada": confianza_combinada,
        "veredicto_final": veredicto
    }
