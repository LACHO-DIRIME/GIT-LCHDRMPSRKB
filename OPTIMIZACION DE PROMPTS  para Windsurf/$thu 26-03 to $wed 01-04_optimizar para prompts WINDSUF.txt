def registrar_sesion(scalar_s: float, tx_count: int, notas: str = ""):
    conn = sqlite3.connect(DB)
    conn.execute(
        "INSERT OR IGNORE INTO sessions "
        "(fecha, dia, hexagrama, scalar_s, tx_count, notas) "
        "VALUES (?, ?, ?, ?, ?, ?)",
        (datetime.now().isoformat(), DIA, HEXAGRAMA, scalar_s, tx_count, notas)
    )
    conn.commit(); conn.close()
    print(f"✅ Sesión {DIA} registrada · S={scalar_s} · TX={tx_count}")

<script>
let tareas=[];
let idCount=0;

function genId(){
  return (++idCount).toString(16).padStart(8,'0').slice(-8);
}

function genRef(dia,slice,cluster){
  const tomo=dia==='$thu'?'A':dia==='$fri'?'B':'C';
  return `[term:${tomo}°${slice}#${cluster.replace('#','')}]`;
}

function planifica(){
  const titulo=document.getElementById('p-titulo').value.trim();
  if(!titulo){alert('titulo requerido');return;}
  const dia=document.getElementById('p-dia').value;
  const cluster=document.getElementById('p-cluster').value;
  const slice=parseInt(document.getElementById('p-slice').value)||90;
  const id=genId();
  const ref=genRef(dia,slice,cluster);
  tareas.push({id,titulo,dia,cluster,slice,estado:'UF[H05]',ref});
  document.getElementById('p-ref').textContent=
    `✦ ref_soberana(): ${ref} · id: ${id}`;
  document.getElementById('p-titulo').value='';
  renderTareas();
  updateStats();
}

function activar(id){
  const t=tareas.find(t=>t.id===id);
  if(t&&t.estado==='UF[H05]'){t.estado='UF[H57]';renderTareas();updateStats();}
}

function resolver(id){
  const t=tareas.find(t=>t.id===id);
  if(t){t.estado='UF[H63]';renderTareas();updateStats();}
}

function renderTareas(){
  const body=document.getElementById('tareas-body');
  if(!tareas.length){
    body.innerHTML='<div style="font-size:.65rem;color:#aaa;padding:10px">'+
      'sin tareas · planifica() para crear TareaSoberana</div>';
    return;
  }
  body.innerHTML=tareas.map(t=>`
    <div class="tl-row">
      <div style="font-size:.6rem;color:#888">${t.id}</div>
      <div>${t.dia}</div>
      <div>${t.titulo.slice(0,35)}${t.titulo.length>35?'…':''}</div>
      <div class="tl-ref">${t.ref}</div>
      <div class="${'st-'+t.estado.replace('UF[','').replace(']','').toLowerCase()}">${t.estado}</div>
      <div>
        ${t.estado==='UF[H05]'?
          `<button class="act-btn" onclick="activar('${t.id}')">activar</button>`:
          t.estado==='UF[H57]'?
          `<button class="res-btn" onclick="resolver('${t.id}')">resolver</button>`:
          '<span style="font-size:.6rem;color:#aaa">H63 ✓</span>'}
      </div>
    </div>`).join('');
}

function updateStats(){
  document.getElementById('st-esp').textContent=
    tareas.filter(t=>t.estado==='UF[H05]').length;
  document.getElementById('st-act').textContent=
    tareas.filter(t=>t.estado==='UF[H57]').length;
  document.getElementById('st-res').textContent=
    tareas.filter(t=>t.estado==='UF[H63]').length;
  document.getElementById('st-con').textContent=
    tareas.filter(t=>t.estado==='UF[H06]').length;
}

renderTareas();

let tareas=[];
let idCount=0;

function genId(){
  return (++idCount).toString(16).padStart(8,'0').slice(-8);
}

function genRef(dia,slice,cluster){
  const tomo=dia==='$thu'?'A':dia==='$fri'?'B':'C';
  return `[term:${tomo}°${slice}#${cluster.replace('#','')}]`;
}

function planifica(){
  const titulo=document.getElementById('p-titulo').value.trim();
  if(!titulo){alert('titulo requerido');return;}
  const dia=document.getElementById('p-dia').value;
  const cluster=document.getElementById('p-cluster').value;
  const slice=parseInt(document.getElementById('p-slice').value)||90;
  const id=genId();
  const ref=genRef(dia,slice,cluster);
  tareas.push({id,titulo,dia,cluster,slice,estado:'UF[H05]',ref});
  document.getElementById('p-ref').textContent=
    `✦ ref_soberana(): ${ref} · id: ${id}`;
  document.getElementById('p-titulo').value='';
  renderTareas();
  updateStats();
}

function activar(id){
  const t=tareas.find(t=>t.id===id);
  if(t&&t.estado==='UF[H05]'){t.estado='UF[H57]';renderTareas();updateStats();}
}

function resolver(id){
  const t=tareas.find(t=>t.id===id);
  if(t){t.estado='UF[H63]';renderTareas();updateStats();}
}

function renderTareas(){
  const body=document.getElementById('tareas-body');
  if(!tareas.length){
    body.innerHTML='<div style="font-size:.65rem;color:#aaa;padding:10px">'+
      'sin tareas · planifica() para crear TareaSoberana</div>';
    return;
  }
  body.innerHTML=tareas.map(t=>`
    <div class="tl-row">
      <div style="font-size:.6rem;color:#888">${t.id}</div>
      <div>${t.dia}</div>
      <div>${t.titulo.slice(0,35)}${t.titulo.length>35?'…':''}</div>
      <div class="tl-ref">${t.ref}</div>
      <div class="${'st-'+t.estado.replace('UF[','').replace(']','').toLowerCase()}">${t.estado}</div>
      <div>
        ${t.estado==='UF[H05]'?
          `<button class="act-btn" onclick="activar('${t.id}')">activar</button>`:
          t.estado==='UF[H57]'?
          `<button class="res-btn" onclick="resolver('${t.id}')">resolver</button>`:
          '<span style="font-size:.6rem;color:#aaa">H63 ✓</span>'}
      </div>
    </div>`).join('');
}

function updateStats(){

    tareas.filter(t=>t.estado==='UF[H05]').length;
  document.getElementById('st-act').textContent=
    tareas.filter(t=>t.estado==='UF[H57]').length;
