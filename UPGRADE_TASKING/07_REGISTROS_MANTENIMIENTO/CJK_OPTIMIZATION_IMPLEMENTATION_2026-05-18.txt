class IntelligentCache:
    \"\"\"Caché inteligente inspirado en DYNAMIC STOCK_CHAIR\"\"\"
    
    def __init__(self, cache_dir='./cache', max_size_mb=100):
        self.cache_dir = cache_dir
        self.max_size_mb = max_size_mb
        self.cache_index = {}
        self._ensure_cache_dir()
    
    def _ensure_cache_dir(self):
        \"\"\"Asegurar que existe directorio de caché\"\"\"
        if not os.path.exists(self.cache_dir):
            os.makedirs(self.cache_dir)
    
    def _get_cache_key(self, data) -> str:
        \"\"\"Generar clave de caché basada en hash\"\"\"
        return hashlib.md5(str(data).encode()).hexdigest()
    
    def get(self, key: str):
        \"\"\"Obtener dato de caché\"\"\"
        cache_file = os.path.join(self.cache_dir, f'{key}.cache')
        if os.path.exists(cache_file):
            try:
                with open(cache_file, 'rb') as f:
                    cached_data = pickle.load(f)
                    # Verificar si no ha expirado (24 horas)
                    if datetime.now() - cached_data['timestamp'] < timedelta(hours=24):
                        return cached_data['data']
                    else:
                        os.remove(cache_file)
            except:
                pass
        return None
    
    def set(self, key: str, data):
        \"\"\"Guardar dato en caché\"\"\"
        cache_file = os.path.join(self.cache_dir, f'{key}.cache')
        cached_data = {
            'data': data,
            'timestamp': datetime.now()
        }
        
        try:
            with open(cache_file, 'wb') as f:
                pickle.dump(cached_data, f)
            
            # Limpiar caché si excede tamaño máximo
            self._cleanup_cache()
        except Exception as e:
            print(f'Error guardando en caché: {e}')
    
    def _cleanup_cache(self):
        \"\"\"Limpiar caché si excede tamaño máximo\"\"\"
        total_size = sum(os.path.getsize(os.path.join(self.cache_dir, f)) 
                        for f in os.listdir(self.cache_dir) if f.endswith('.cache'))
        
        if total_size > self.max_size_mb * 1024 * 1024:
            # Eliminar archivos más antiguos
            cache_files = [(os.path.getmtime(os.path.join(self.cache_dir, f)), 
                            os.path.join(self.cache_dir, f))
                           for f in os.listdir(self.cache_dir) if f.endswith('.cache')]
            cache_files.sort()
            
            # Eliminar hasta estar bajo el límite
            for _, file_path in cache_files:
                os.remove(file_path)
                total_size -= os.path.getsize(file_path)
                if total_size < self.max_size_mb * 1024 * 1024 * 0.8:
                    break
