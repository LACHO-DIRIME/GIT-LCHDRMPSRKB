import sys
import os
from typing import Dict, List, Any, Optional
from dataclasses import dataclass

class UnicodeCJKBridge:
    """Puente entre sistema Unicode y CJK optimizado"""
    
    def __init__(self):
        self.unicode_path = "/media/Personal/PLANERAI/LACHO/_DIRIME/UNICODE"
        self.cjk_path = "/media/Personal/PLANERAI/LACHO/Lenguaje_CJK(SOBERANO)"
        self.bridge_active = False
        
        # Verificar disponibilidad de sistemas
        self._check_system_availability()
    
    def _check_system_availability(self):
        """Verificar disponibilidad de sistemas"""
        unicode_available = os.path.exists(self.unicode_path)
        cjk_available = os.path.exists(self.cjk_path)
        
        if unicode_available and cjk_available:
            self.bridge_active = True
        else:
            print(f"Bridge no disponible: Unicode={unicode_available}, CJK={cjk_available}")
    
    def bridge_unicode_to_cjk(self, unicode_data: Dict[str, Any]) -> Dict[str, Any]:
        """Puentear datos Unicode a sistema CJK"""
        if not self.bridge_active:
            return {
                'status': 'bridge_unavailable',
                'message': 'Bridge Unicode-CJK no disponible'
            }
        
        # Extraer información Unicode relevante
        cjk_relevant_data = {
            'cjk_blocks': self._extract_cjk_blocks(unicode_data),
            'cjk_scripts': self._extract_cjk_scripts(unicode_data),
            'cjk_languages': self._extract_cjk_languages(unicode_data),
            'processing_metadata': {
                'bridge_timestamp': time.time(),
                'unicode_source': 'lacho_unicode',
                'cjk_target': 'optimized_cjk'
            }
        }
        
        return {
            'status': 'bridge_success',
            'cjk_data': cjk_relevant_data,
            'unicode_data': unicode_data
        }
    
    def _extract_cjk_blocks(self, unicode_data: Dict[str, Any]) -> List[str]:
        """Extraer bloques CJK relevantes"""
        cjk_blocks = []
        
        if 'unicode_blocks' in unicode_data:
            blocks_found = unicode_data['unicode_blocks'].get('blocks_found', {})
            for block_name in blocks_found.keys():
                if any(cjk_term in block_name.lower() for cjk_term in ['cjk', 'hiragana', 'katakana', 'hangul']):
                    cjk_blocks.append(block_name)
        
        return cjk_blocks
    
    def _extract_cjk_scripts(self, unicode_data: Dict[str, Any]) -> List[str]:
        """Extraer scripts CJK relevantes"""
        cjk_scripts = []
        
        if 'scripts_detected' in unicode_data:
            for script in unicode_data['scripts_detected']:
                if hasattr(script, 'value'):
                    script_name = script.value
                else:
                    script_name = str(script)
                
                if any(cjk_term in script_name.lower() for cjk_term in ['han', 'hiragana', 'katakana', 'hangul']):
                    cjk_scripts.append(script_name)
        
        return cjk_scripts
    
    def _extract_cjk_languages(self, unicode_data: Dict[str, Any]) -> List[str]:
        """Extraer idiomas CJK relevantes"""
        cjk_languages = []
        
        if 'languages_detected' in unicode_data:
            cjk_langs = ['chinese', 'japanese', 'korean']
            for lang in unicode_data['languages_detected']:
                if lang in cjk_langs:
                    cjk_languages.append(lang)
        
        return cjk_languages
    
    def get_bridge_status(self) -> Dict[str, Any]:
        """Obtener estado del bridge"""
        return {
            'bridge_active': self.bridge_active,
            'unicode_path': self.unicode_path,
            'cjk_path': self.cjk_path,
            'unicode_available': os.path.exists(self.unicode_path),
            'cjk_available': os.path.exists(self.cjk_path)
        }
