    def identify_unicode_blocks(self, text: str) -> Dict[UnicodeBlock, List[str]]:
        """Identificar bloques Unicode en texto"""
        start_time = time.time()
        
        blocks_found = defaultdict(list)
        
        for char in text:
            char_code = ord(char)
            
            # Verificar caché
            if char in self.character_cache:
                block = self.character_cache[char]
                self.processing_stats['cache_hits'] += 1
            else:
                block = self._identify_block(char_code)
                self.character_cache[char] = block
                self.processing_stats['cache_misses'] += 1
            
            if block:
                blocks_found[block].append(char)
                self.processing_stats['blocks_detected'][block] += 1
        
        self.processing_stats['total_processed'] += len(text)
        
        end_time = time.time()
        
        return {
            'blocks_found': dict(blocks_found),
            'processing_time': end_time - start_time,
            'text_length': len(text),
            'unique_blocks': len(blocks_found)
        }
    
    def _identify_block(self, char_code: int) -> Optional[UnicodeBlock]:
        """Identificar bloque Unicode de un carácter"""
        for block, range_info in self.unicode_blocks.items():
            if range_info.start <= char_code <= range_info.end:
                return block
        return None
    
    def get_unicode_statistics(self) -> Dict[str, Union[int, float, Dict]]:
        """Obtener estadísticas Unicode completas"""
        total_chars = sum(range_info.count for range_info in self.unicode_blocks.values())
        
        return {
            'total_blocks': len(self.unicode_blocks),
            'total_characters_supported': total_chars,
            'blocks_info': {
                block.value: {
                    'range': f"0x{range_info.start:04X}-0x{range_info.end:04X}",
                    'count': range_info.count,
                    'description': range_info.description,
                    'sample_chars': range_info.sample_chars
                }
                for block, range_info in self.unicode_blocks.items()
            },
            'processing_stats': self.processing_stats.copy(),
            'cache_hit_rate': self.processing_stats['cache_hits'] / max(
                self.processing_stats['cache_hits'] + self.processing_stats['cache_misses'], 1
            )
        }
    
    def process_multilingual_text(self, text: str) -> Dict[str, any]:
        """Procesar texto multilingüe completo"""
        start_time = time.time()
        
        # Identificar bloques
        blocks = self.identify_unicode_blocks(text)
        
        # Análisis de composición
        composition = self._analyze_text_composition(text)
        
        # Normalización Unicode
        normalized = self._normalize_text(text)
        
        end_time = time.time()
        
        return {
            'original_text': text,
            'unicode_blocks': blocks,
            'composition': composition,
            'normalized_text': normalized,
            'processing_time': end_time - start_time,
            'statistics': self.get_unicode_statistics()
        }
    
    def _analyze_text_composition(self, text: str) -> Dict[str, any]:
        """Analizar composición del texto"""
        composition = {
            'total_length': len(text),
            'unique_characters': len(set(text)),
            'alphabetic': sum(1 for c in text if c.isalpha()),
            'numeric': sum(1 for c in text if c.isdigit()),
            'punctuation': sum(1 for c in text if not c.isalnum() and not c.isspace()),
            'whitespace': sum(1 for c in text if c.isspace()),
            'control_chars': sum(1 for c in text if ord(c) < 32)
        }
        
        return composition
    
    def _normalize_text(self, text: str) -> str:
        """Normalizar texto Unicode"""
        # Normalización NFC (Canonical Decomposition followed by Canonical Composition)
        normalized = unicodedata.normalize('NFC', text)
        return normalized

        start_time = time.time()
        
        # Verificar caché
        text_hash = hashlib.md5(text.encode()).hexdigest()
        with self.lock:
            if text_hash in self.cache:
                self.stats['cache_hits'] += 1
                cached_result = self.cache[text_hash]
                return UnicodeAnalysis(**cached_result)
            else:
                self.stats['cache_misses'] += 1
        
        # Detectar scripts
        scripts_detected = self._detect_scripts(text)
        
        # Detectar bloques
        blocks_detected = self._detect_blocks(text)
        
        # Detectar idiomas
        languages_detected = self._detect_languages(text)
        
        # Calcular complejidad
        complexity_score = self._calculate_complexity(text, scripts_detected)
        
        end_time = time.time()
        processing_time = end_time - start_time
        
        # Crear análisis
        analysis = UnicodeAnalysis(
            text=text,
            scripts_detected=scripts_detected,
            blocks_detected=blocks_detected,
            character_count=len(text),
            unique_characters=len(set(text)),
            languages_detected=languages_detected,
            complexity_score=complexity_score,
            processing_time=processing_time
        )
        
        # Guardar en caché
        with self.lock:
            self.cache[text_hash] = {
                'text': analysis.text,
                'scripts_detected': analysis.scripts_detected,
                'blocks_detected': analysis.blocks_detected,
                'character_count': analysis.character_count,
                'unique_characters': analysis.unique_characters,
                'languages_detected': analysis.languages_detected,
                'complexity_score': analysis.complexity_score,
                'processing_time': analysis.processing_time
            }
        
        # Actualizar estadísticas
        with self.lock:
            self.stats['texts_processed'] += 1
            self.stats['characters_processed'] += len(text)
            self.stats['processing_time_total'] += processing_time
        
        return analysis
    
    def _detect_scripts(self, text: str) -> List[UnicodeScript]:
        """Detectar scripts Unicode en texto"""
        scripts = set()
        
        for script, pattern in self.script_patterns.items():
            if re.search(pattern, text):
                scripts.add(script)
        
        return list(scripts)
    
    def _detect_blocks(self, text: str) -> List[str]:
        """Detectar bloques Unicode en texto"""
        blocks = set()
        
        for char in text:
            char_code = ord(char)
            block_name = unicodedata.name(char, 'UNKNOWN').split(' ')[0]
            blocks.add(block_name)
        
        return list(blocks)
    
    def _detect_languages(self, text: str) -> List[str]:
        """Detectar idiomas en texto"""
        languages = set()
        
        for language, patterns in self.language_patterns.items():
            confidence = 0
            for pattern_type, pattern in patterns.items():
                matches = len(re.findall(pattern, text))
                confidence += matches
            
            if confidence > 0:
                languages.add(language)
        
        return list(languages)
    
    def _calculate_complexity(self, text: str, scripts: List[UnicodeScript]) -> float:
        """Calcular score de complejidad del texto"""
        base_score = len(set(text)) / max(len(text), 1)
        
        # Penalizar scripts complejos
        script_weights = {
            UnicodeScript.LATIN: 1.0,
            UnicodeScript.HAN: 2.0,
            UnicodeScript.HIRAGANA: 1.5,
            UnicodeScript.KATAKANA: 1.5,
            UnicodeScript.HANGUL: 1.8,
            UnicodeScript.ARABIC: 1.6,
            UnicodeScript.CYRILLIC: 1.4,
            UnicodeScript.GREEK: 1.3,
            UnicodeScript.HEBREW: 1.5,
            UnicodeScript.THAI: 1.7,
            UnicodeScript.DEVANAGARI: 1.8,
            UnicodeScript.BENGALI: 1.8,
            UnicodeScript.TAMIL: 1.8,
            UnicodeScript.TELUGU: 1.8
        }
        
        script_complexity = sum(script_weights.get(script, 1.0) for script in scripts)
        
        return base_score * script_complexity
    
    def process_batch_unicode(self, texts: List[str]) -> List[UnicodeAnalysis]:
        """Procesar batch de textos Unicode"""
        start_time = time.time()
        
        futures = []
        for text in texts:
            future = self.executor.submit(self.analyze_text_unicode, text)
            futures.append(future)
        
        results = []
        for future in futures:
            try:
                result = future.result(timeout=10.0)
                results.append(result)
            except Exception as e:
                print(f'Error en procesamiento Unicode: {e}')
        
        end_time = time.time()
        
        return results
    
    def get_processing_stats(self) -> Dict[str, Any]:
        """Obtener estadísticas de procesamiento"""
        with self.lock:
            cache_hit_rate = self.stats['cache_hits'] / max(
                self.stats['cache_hits'] + self.stats['cache_misses'], 1
            )
            
            return {
                'texts_processed': self.stats['texts_processed'],
                'characters_processed': self.stats['characters_processed'],
                'cache_hit_rate': cache_hit_rate,
                'average_processing_time': self.stats['processing_time_total'] / max(self.stats['texts_processed'], 1),
                'cache_size': len(self.cache),
                'stats': self.stats.copy()
            }
